
CREATE OR REPLACE VIEW vista_ventas_apiario AS
SELECT
    u.departamento,
    u.municipio,
    a.nombre                           AS apiario,
    p.nombre                           AS productor,
    COUNT(DISTINCT pe.id_pedido)       AS total_pedidos,
    SUM(dp.cantidad * dp.precio_venta) AS monto_total_COP,
    AVG(dp.precio_venta)               AS precio_promedio
FROM   detalle_pedido dp
JOIN   pedido         pe ON pe.id_pedido   = dp.id_pedido
JOIN   productor      p  ON p.id_productor = pe.id_productor
JOIN   apiario        a  ON a.id_productor = p.id_productor
JOIN   ubicacion      u  ON u.id_ubicacion = a.id_ubicacion
GROUP  BY u.departamento, u.municipio, a.nombre, p.nombre
HAVING SUM(dp.cantidad * dp.precio_venta) > 40000
ORDER  BY monto_total_COP DESC;


SELECT * FROM vista_ventas_apiario;

WHERE departamento = 'Antioquia';
